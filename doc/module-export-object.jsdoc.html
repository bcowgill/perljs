    /**
     * A module that exports a variable other than using exports or module.exports.
     * @exports JSDocModule/ExportVar

    var JSDocExportVar = {};

    /** Say hello. */
    JSDocExportVar.sayHello = function() {
        return "Hello world";
    };

    return JSDocExportVar;
});
